INCLUDE_DIRECTORIES(${WZDFTPD_SOURCE_DIR}
	${WZDFTPD_SOURCE_DIR}/libwzd)

LINK_DIRECTORIES (${WZDFTPD_BINARY_DIR}/libwzd)

ADD_EXECUTABLE (siteconfig siteconfig.c)

# Link the executable to the libwzd-core library.
TARGET_LINK_LIBRARIES (siteconfig libwzd)

IF (NOT SKIP_INSTALL_RPATH)
  SET_TARGET_PROPERTIES(siteconfig PROPERTIES INSTALL_RPATH "${CMAKE_INSTALL_PREFIX}/lib")
ENDIF (NOT SKIP_INSTALL_RPATH)
INSTALL(TARGETS siteconfig RUNTIME DESTINATION bin)
