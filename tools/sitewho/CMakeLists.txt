INCLUDE_DIRECTORIES(${WZDFTPD_SOURCE_DIR}
	${WZDFTPD_SOURCE_DIR}/libwzd)

LINK_DIRECTORIES (${WZDFTPD_BINARY_DIR}/libwzd)

ADD_EXECUTABLE (sitewho sitewho.c)

# Link the executable to the libwzd-core library.
TARGET_LINK_LIBRARIES (sitewho libwzd)

IF (NOT SKIP_INSTALL_RPATH)
  SET_TARGET_PROPERTIES(sitewho PROPERTIES INSTALL_RPATH "${CMAKE_INSTALL_PREFIX}/lib")
ENDIF (NOT SKIP_INSTALL_RPATH)
INSTALL(TARGETS sitewho RUNTIME DESTINATION ${BIN_INSTALL_PATH})
