INCLUDE_DIRECTORIES(${WZDFTPD_SOURCE_DIR}
	${WZDFTPD_SOURCE_DIR}/libwzd)

LINK_DIRECTORIES (${WZDFTPD_BINARY_DIR}/libwzd)

ADD_EXECUTABLE (sitewho sitewho.c)

TARGET_LINK_LIBRARIES (sitewho libwzd)

INSTALL(TARGETS sitewho RUNTIME DESTINATION bin)
